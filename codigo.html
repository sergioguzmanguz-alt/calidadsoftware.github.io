Ejemplo de Código Limpio


function calcularPromedio(notas) {
  if (!notas.length) return 0;
  const suma = notas.reduce((a, b) => a + b, 0);
  return suma / notas.length;
}

El código limpio es legible, reutilizable y fácil de mantener.
